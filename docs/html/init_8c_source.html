48bool wu_signal_validate(const WU_Signal* signal) {
49 if (!signal) return false;
50
51 // Validate price is positive
52 if (signal->price <= 0.0) {
53 return false;
54 }
55
56 // Validate quantity is non-negative (can be 0 for HOLD signals)
57 if (signal->quantity < 0.0) {
58 return false;
59 }
60
61 // Validate side is a known value
62 if (signal->side != WU_SIDE_HOLD &&
63 signal->side != WU_SIDE_BUY &&
64 signal->side != WU_SIDE_SELL) {
65 return false;
66 }
67
68 return true;
