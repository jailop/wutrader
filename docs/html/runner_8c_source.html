17static bool validate_input_type(WU_DataType reader_type, WU_DataType expected_type) {
18 // Exact match is always valid
19 if (reader_type == expected_type) {
20 return true;
21 }
22
23 // Candle can be converted to Single (using close price)
24 if (reader_type == WU_DATA_TYPE_CANDLE && expected_type == WU_DATA_TYPE_SINGLE_VALUE) {
25 return true;
26 }
27
28 // Trade can be converted to Single (using trade price)
29 if (reader_type == WU_DATA_TYPE_TRADE && expected_type == WU_DATA_TYPE_SINGLE_VALUE) {
30 return true;
31 }
32
33 return false;
34}

51static void run_backtest(WU_Runner runner, bool verbose) {
52 if (!runner) return;
53
54 WU_Portfolio portfolio = runner->portfolio;
55 WU_Strategy strategy = runner->strategy;
56
57 // Allocate input buffers dynamically based on number of readers
58 void** input_buffers = malloc(sizeof(void*) * runner->num_readers);
59 WU_Single* converted = malloc(sizeof(WU_Single) * runner->num_readers);
60
61 if (!input_buffers || !converted) {
62 fprintf(stderr, "Error: Failed to allocate input buffers\n");
63 free(input_buffers);
64 free(converted);
65 return;
66 }
67
68 if (verbose) {
69 printf("Starting backtest with %d input(s)...\n", runner->num_readers);
70 }
71
72 int iteration = 0;
73 while (true) {
74 // Read from all readers in lockstep
75 bool all_valid = true;
76
77 for (int i = 0; i < runner->num_readers; i++) {
78 void* data = wu_reader_next(runner->readers[i]);
79
80 if (!data) {
81 all_valid = false;
82 break;
83 }
84
85 // Determine actual data type from the data itself
86 WU_DataType actual_type = ((WU_Candle*)data)->data_type;
87 WU_DataType expected_type = strategy->input_types[i];
88
89 // Apply automatic type conversion if needed
90 if (actual_type == WU_DATA_TYPE_CANDLE &&
91 expected_type == WU_DATA_TYPE_SINGLE_VALUE) {
92 // Convert Candle to Single using close price
93 converted[i] = wu_candle_to_single_value((WU_Candle*)data);
94 input_buffers[i] = &converted[i];
95 }
96 else if (actual_type == WU_DATA_TYPE_TRADE &&
97 expected_type == WU_DATA_TYPE_SINGLE_VALUE) {
98 // Convert Trade to Single using trade price
99 converted[i] = wu_trade_to_single_value((WU_Trade*)data);
100 input_buffers[i] = &converted[i];
101 }
102 else if (actual_type == expected_type) {
103 // Direct pass-through, no conversion needed
104 input_buffers[i] = data;
105 }
106 else {
107 // Type mismatch that cannot be converted
108 fprintf(stderr, "Error at iteration %d: Input %d has type %d but strategy expects type %d\n",
109 iteration, i, actual_type, expected_type);
110 all_valid = false;
111 break;
112 }
113 }
114
115 // Stop if any reader finished or had an error
116 if (!all_valid) {
117 break;
118 }
119
120 // Update strategy with all inputs
121 wu_strategy_update(strategy, (const void**)input_buffers);
122
123 // Update portfolio with signals from strategy
124 wu_portfolio_update(portfolio, strategy->signal_buffer);
125
126 // Print verbose output if requested
127 if (verbose && iteration % 100 == 0) {
128 printf("Iteration %d | Signals: %d | Value: %.2f | P&L: %.2f\n",
129 iteration,
130 strategy->num_outputs,
131 wu_portfolio_value(portfolio),
132 wu_portfolio_pnl(portfolio));
133 }
134
135 iteration++;
136 }
137
138 if (verbose) {
139 printf("Backtest completed after %d iterations\n", iteration);
140 }
141
142 // Free allocated buffers
143 free(input_buffers);
144 free(converted);
145}

147WU_Runner wu_runner_new(
148 WU_Portfolio portfolio,
149 WU_Strategy strategy,
150 WU_Reader readers[]
151) {
152 // Validate inputs
153 if (!portfolio || !strategy || !readers) {
154 fprintf(stderr, "Error: NULL argument passed to wu_runner_new\n");
155 return NULL;
156 }
157
158 int num_readers = wu_strategy_num_inputs(strategy);
159
160 // Validate each reader's data type compatibility
161 for (int i = 0; i < num_readers; i++) {
162 if (!readers[i]) {
163 fprintf(stderr, "Error: Reader %d is NULL\n", i);
164 return NULL;
165 }
166
167 WU_DataType reader_type = get_reader_data_type(readers[i]);
168 WU_DataType expected_type = strategy->input_types[i];
169
170 if (!validate_input_type(reader_type, expected_type)) {
171 fprintf(stderr, "Error: Reader %d produces type %d but strategy expects type %d\n",
172 i, reader_type, expected_type);
173 return NULL;
174 }
175 }
176
177 // Allocate runner
178 WU_Runner runner = malloc(sizeof(struct WU_Runner_));
179 if (!runner) {
180 fprintf(stderr, "Error: Failed to allocate runner\n");
181 return NULL;
182 }
183
184 // Allocate and copy reader array
185 runner->readers = malloc(sizeof(WU_Reader) * num_readers);
186 if (!runner->readers) {
187 fprintf(stderr, "Error: Failed to allocate reader array\n");
188 free(runner);
189 return NULL;
190 }
191
192 memcpy(runner->readers, readers, sizeof(WU_Reader) * num_readers);
193
194 // Initialize runner
195 runner->portfolio = portfolio;
196 runner->strategy = strategy;
197 runner->num_readers = num_readers;
198 runner->run = run_backtest;
199
200 return runner;
201}

203void wu_runner_free(WU_Runner runner) {
204 if (!runner) return;
205
206 // Free portfolio
207 wu_portfolio_delete(runner->portfolio);
208
209 // Free strategy
210 wu_strategy_delete(runner->strategy);
211
212 // Free all readers
213 for (int i = 0; i < runner->num_readers; i++) {
214 wu_reader_delete(runner->readers[i]);
215 }
216 free(runner->readers);
217
218 // Free runner itself
219 free(runner);
220}
