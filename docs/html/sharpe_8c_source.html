6static double wu_sharpe_ratio_update(WU_SharpeRatio self,
7 WU_PerformanceUpdate perf) {
8 /* On first observed perf, record start time but still record the return
9 relative to the initial value so that returns are (value - initial)/initial */
10 if (self->count == 0) {
11 self->start_time = perf.timestamp.mark;
12 self->time_unit = perf.timestamp.units;
13 }
14 double ret = (perf.portfolio_value - self->initial_value) / self->initial_value;
15 self->end_time = perf.timestamp.mark;
16 self->count++;
17 double mean = wu_indicator_update(self->mean, ret);
18 double stdev = wu_indicator_update(self->stdev, ret);
19 if (isnan(stdev) || stdev <= 0.0) {
20 self->value = NAN;
21 return self->value;
22 }
23 double annualization_factor = wu_annualization_factor(perf.timestamp.units);
24 double periods_elapsed = self->end_time - self->start_time;
25 if (periods_elapsed <= 0) periods_elapsed = 1;
26 double periods_per_year = annualization_factor / periods_elapsed * self->count;
27 if (periods_per_year <= 0) periods_per_year = 1;
28 /* Annualize stdev and compute Sharpe: (mean_per_period - rf_per_period)/stdev_per_period * sqrt(periods_per_year)
29 Note: return_stats stores per-update returns (relative to initial), so mean/stdev are per-update stats. */
30 double annualized_stdev = stdev * sqrt(periods_per_year);
31 double per_period_rf = self->risk_free_rate / periods_per_year;
32 self->value = (mean - per_period_rf) / annualized_stdev;
33 return self->value;
34}
