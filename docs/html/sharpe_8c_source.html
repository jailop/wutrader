5// Annualization factor based on time unit

7static double wu_annualization_factor(WU_TimeUnit unit) {
8 switch (unit) {
9 case WU_TIME_UNIT_SECONDS: return 365.25 * 24 * 3600; // seconds per year
10 case WU_TIME_UNIT_MILLIS: return 365.25 * 24 * 3600 * 1000; // millis per year
