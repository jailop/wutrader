47static void stats_update_position(struct WU_PortfolioStats_* stats, int asset_index,
48 const char* symbol, double quantity, double value, double last_price) {
49 if (!stats) return;
50 if (asset_index < 0) return;
51
52 // Expand capacity if needed
53 if (asset_index >= stats->capacity) {
54 int new_capacity = asset_index + 1;
55 stats->symbols = realloc(stats->symbols, new_capacity * sizeof(char*));
56 stats->quantities = realloc(stats->quantities, new_capacity * sizeof(double));
57 stats->values = realloc(stats->values, new_capacity * sizeof(double));
58 stats->last_prices = realloc(stats->last_prices, new_capacity * sizeof(double));
59
60 for (int i = stats->capacity; i < new_capacity; i++) {
61 stats->symbols[i] = NULL;
62 stats->quantities[i] = 0.0;
63 stats->values[i] = 0.0;
64 stats->last_prices[i] = 0.0;
65 }
66 stats->capacity = new_capacity;
67 }
68
69 if (asset_index >= stats->num_assets)
70 stats->num_assets = asset_index + 1;
71
72 // Update or set symbol
73 if (symbol && (!stats->symbols[asset_index] ||
74 strcmp(stats->symbols[asset_index], symbol) != 0)) {
75 free(stats->symbols[asset_index]);
76 stats->symbols[asset_index] = wu_strdup(symbol);
77 }
78
79 stats->quantities[asset_index] = quantity;
80 stats->values[asset_index] = value;
81 stats->last_prices[asset_index] = last_price;
82}
