39static void stats_record_trade(struct WU_PortfolioStats_* stats, double pnl,
40 WU_CloseReason reason) {
41 if (!stats) return;
42
43 stats->total_trades++;
44 stats->accum_pnl += pnl;
45
46 // Update PnL statistics
47 if (stats->pnl_stats)
48 stats->pnl_stats->update(stats->pnl_stats, pnl);
49
50 if (pnl > 0) {
51 stats->winning_trades++;
52 stats->total_profit += pnl;
53 if (pnl > stats->max_win)
54 stats->max_win = pnl;
55 } else if (pnl < 0) {
56 stats->losing_trades++;
57 stats->total_loss += pnl;
58 if (pnl < stats->max_loss)
59 stats->max_loss = pnl;
60 }
61
62 if (reason == WU_CLOSE_REASON_STOP_LOSS)
63 stats->stop_loss_exits++;
64 else if (reason == WU_CLOSE_REASON_TAKE_PROFIT)
65 stats->take_profit_exits++;
66}

68static void stats_update_position(struct WU_PortfolioStats_* stats, int asset_index,
69 const char* symbol, double quantity, double value, double last_price) {
70 if (!stats) return;
71 if (asset_index < 0) return;
72
73 // Expand capacity if needed
74 if (asset_index >= stats->capacity) {
75 int new_capacity = asset_index + 1;
76 stats->symbols = realloc(stats->symbols, new_capacity * sizeof(char*));
77 stats->quantities = realloc(stats->quantities, new_capacity * sizeof(double));
78 stats->values = realloc(stats->values, new_capacity * sizeof(double));
79 stats->last_prices = realloc(stats->last_prices, new_capacity * sizeof(double));
80
81 for (int i = stats->capacity; i < new_capacity; i++) {
82 stats->symbols[i] = NULL;
83 stats->quantities[i] = 0.0;
84 stats->values[i] = 0.0;
85 stats->last_prices[i] = 0.0;
86 }
87 stats->capacity = new_capacity;
88 }
89
90 if (asset_index >= stats->num_assets)
91 stats->num_assets = asset_index + 1;
92
93 // Update or set symbol
94 if (symbol && (!stats->symbols[asset_index] ||
95 strcmp(stats->symbols[asset_index], symbol) != 0)) {
96 free(stats->symbols[asset_index]);
97 stats->symbols[asset_index] = wu_strdup(symbol);
98 }
99
100 stats->quantities[asset_index] = quantity;
101 stats->values[asset_index] = value;
102 stats->last_prices[asset_index] = last_price;
